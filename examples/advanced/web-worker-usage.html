üíª Code Example

// Using jtcsv with Web Workers
import { createWorkerPool } from 'jtcsv/browser/workers';

// Create a pool of 4 workers
const workerPool = createWorkerPool('./csv-parser.worker.js', {
  maxWorkers: 4,
  idleTimeout: 30000
});

// Process CSV in background
async function processLargeCsv(csvData) {
  const startTime = performance.now();
  
  // Submit task to worker pool
  const result = await workerPool.execute({
    type: 'parseCsv',
    csv: csvData,
    options: {
      hasHeaders: true,
      parseNumbers: true,
      parseBooleans: true
    }
  });
  
  const endTime = performance.now();
  console.log(`Processed ${result.data.length} rows in ${endTime - startTime}ms`);
  
  return result.data;
}

// Clean up when done
workerPool.terminate();
